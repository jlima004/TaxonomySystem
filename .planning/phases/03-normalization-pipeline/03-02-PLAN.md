```
import { normalizeUnicode } from './normalize_unicode.ts'
import { normalizeCase } from './normalize_case.ts'
import { normalizeSeparators } from './normalize_separators.ts'
import { removePunctuation } from './remove_punctuation.ts'
import { collapseUnderscores } from './collapse_underscores.ts'
import { trimUnderscores } from './trim_underscores.ts'
import { singularize } from './singularize.ts'

const pipeline = [
  normalizeUnicode,
  normalizeCase,
  normalizeSeparators,
  removePunctuation,
  collapseUnderscores,
  trimUnderscores,
  singularize,
] as const

export const normalizeDescriptor = (input: string): string => {
  return pipeline.reduce((acc, fn) => fn(acc), input)
}
```

ORDEM CANÔNICA OBRIGATÓRIA (D-22):
1. `normalizeUnicode()` — NFD decomposition, diacritic removal, ligature expansion
2. `normalizeCase()` — lowercase
3. `normalizeSeparators()` — spaces, hyphens, slashes → `_`
4. `removePunctuation()` — strip non-alphanumeric/non-underscore
5. `collapseUnderscores()` — multiple `_` → single `_`
6. `trimUnderscores()` — remove leading/trailing `_`
7. `singularize()` — plural → singular (MUST run on already-separated tokens)

NOTA: `singularize` DEVE ser o último step porque depende dos tokens já separados por underscores para fazer singularização token-by-token.

CONTRATO: `singularize()` only accepts normalized snake_case tokens. Input deve ter passado por normalizeCase + normalizeSeparators + removePunctuation + collapseUnderscores + trimUnderscores.

CONTRATO EMPTY OUTPUT: Empty normalized descriptors são outputs válidos. `normalizeDescriptor("!!!")` → `""`. Validação de "descriptor descartável" acontece downstream — o normalizer NUNCA lança ou retorna sentinelas como `"unknown"`. Isso preserva idempotência e pureza.

```
import { describe, it, expect } from 'vitest'
import { normalizeDescriptor } from '../../normalizer/normalize_descriptor.js'

describe('Normalization convergence — canonical collision', () => {
  it('fresh-green, fresh green, fresh/green → same output', () => {
    const variants = ['fresh-green', 'fresh green', 'fresh/green', 'fresh_green', 'FRESH GREEN', 'Fresh-Green']
    const canonical = normalizeDescriptor(variants[0])
    for (const v of variants) {
      expect(normalizeDescriptor(v), `input: "${v}"`).toBe(canonical)
    }
    expect(canonical).toBe('fresh_green')
  })

  it('multiple separator variants converge', () => {
    const variants = ['fresh---green', 'fresh   green', 'fresh///green', 'fresh_-_green', 'fresh___green']
    const canonical = normalizeDescriptor(variants[0])
    for (const v of variants) {
      expect(normalizeDescriptor(v), `input: "${v}"`).toBe(canonical)
    }
    expect(canonical).toBe('fresh_green')
  })

  it('case + separator + punctuation all converge', () => {
    const variants = ['Orange Blossom', 'orange-blossom', 'ORANGE_BLOSSOM', 'orange!!!blossom', '  orange   blossom  ']
    const canonical = normalizeDescriptor(variants[0])
    for (const v of variants) {
      expect(normalizeDescriptor(v), `input: "${v}"`).toBe(canonical)
    }
    expect(canonical).toBe('orange_blossom')
  })

  it('diacritics + ligatures + apostrophe converge to same base', () => {
    const variants = ['Coeur', 'cœur', 'COEUR', 'coeur', "coeur d'aldehyde"]
    const canonical = normalizeDescriptor(variants[0])
    for (const v of variants.slice(0, 4)) {
      expect(normalizeDescriptor(v), `input: "${v}"`).toBe(canonical)
    }
    expect(canonical).toBe('coeur')
    expect(normalizeDescriptor("coeur d'aldehyde")).toBe('coeur_d_aldehyde')
  })

  it('plural variants converge to singular', () => {
    expect(normalizeDescriptor('Woods')).toBe(normalizeDescriptor('wood'))
    expect(normalizeDescriptor('WOODS')).toBe(normalizeDescriptor('wood'))
    expect(normalizeDescriptor('woods')).toBe(normalizeDescriptor('wood'))
    expect(normalizeDescriptor('Mosses')).toBe(normalizeDescriptor('moss'))
  })

  it('protected terminals are NOT modified', () => {
    expect(normalizeDescriptor('gas')).toBe('gas')
    expect(normalizeDescriptor('citrus')).toBe('citrus')
    expect(normalizeDescriptor('iris')).toBe('iris')
    expect(normalizeDescriptor('anis')).toBe('anis')
    expect(normalizeDescriptor('analysis')).toBe('analysis')
    expect(normalizeDescriptor('thesis')).toBe('thesis')
    expect(normalizeDescriptor('basis')).toBe('basis')
    expect(normalizeDescriptor('crisis')).toBe('crisis')
  })

  it('empty variants all converge to empty string', () => {
    const empties = ['', '   ', '!!!', '___', '---', '/ / /']
    for (const e of empties) {
      expect(normalizeDescriptor(e), `input: "${e}"`).toBe('')
    }
  })
})
```

```
import { describe, it, expect } from 'vitest'
import { normalizeDescriptor } from '../../normalizer/normalize_descriptor.js'
import { performance } from 'node:perf_hooks'

describe('Normalization Pipeline Benchmark', () => {
  it('should normalize 100k descriptors under threshold', () => {
    const testInputs = [
      'Fresh Green',
      'ylang-ylang',
      'Mosses',
      'orange blossom',
      'Cœur d\'Aldehyde',
      'fresh---green///woody',
      'Aldehydes_C12',
      'patchouli (dark)',
      'lily of the valley',
      'already_snake_case',
    ]

    const iterations = 100_000
    const start = performance.now()

    for (let i = 0; i < iterations; i++) {
      normalizeDescriptor(testInputs[i % testInputs.length])
    }

    const elapsed = performance.now() - start

    // Threshold: 100k normalizações em < 5000ms (generoso para CI)
    // Em máquinas modernas espera-se < 500ms localmente.
    // Threshold alto evita flaky tests em CI (containers, throttled CPUs, GitHub Actions).
    // O console.log abaixo permite tracking de regressão manual.
    expect(elapsed).toBeLessThan(5000)

    // Log para visibilidade e regression tracking
    console.log(`100k normalizations: ${elapsed.toFixed(2)}ms (${(iterations / elapsed * 1000).toFixed(0)} ops/sec)`)
  })
})
```

NOTA: O threshold de 5000ms é intencionalmente generoso para CI environments. O valor real em máquinas locais deve ser < 500ms (Unicode normalize + 100k iterações em Node/Vitest tem overhead). Se o benchmark começar a falhar, é sinal de regression severa. Para monitoring preciso, usar ops/sec logging em CI artifacts ao invés de hard assertion.
